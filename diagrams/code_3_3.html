@celery_app.task(name="tasks.send_notification", bind=True, max_retries=3)
def send_notification_task(self, return_request_id: int, new_status: str):
    template = STATUS_MESSAGES.get(new_status)
    if not template:
        return {"skipped": True}

    session = SyncSessionLocal()
    rr = session.get(ReturnRequest, return_request_id)

    contact, channel = "", "email"
    if rr.client.phone:
        contact, channel = rr.client.phone, "sms"
    elif rr.client.email:
        contact, channel = rr.client.email, "email"

    message = template.format(number=rr.number)

    cfg = {s.key: s.value for s in session.execute(select(AppSetting)).scalars()}
    if channel == "sms":
        status = send_sms(cfg, contact, message)
    else:
        status = send_email(cfg, contact, subject, message)
    sent = status in ("sent", "simulated")

    session.add(Notification(
        return_request_id=rr.id, recipient_contact=contact,
        channel=channel, message=message, is_sent=sent,
        sent_at=datetime.now(timezone.utc) if sent else None,
    ))
    session.commit()
    return {"channel": channel, "status": status}
